    Create `src/lib/retell/webhook-verify.ts`:
    - Use Node.js built-in `crypto.createHmac('sha256', apiKey)` to verify signature
    - Retell signs with `HMAC-SHA256(body, apiKey)` and sends result as hex in `x-retell-signature`
    - Implement constant-time comparison using `crypto.timingSafeEqual` to prevent timing attacks
    ```typescript
    import crypto from 'crypto'

    export function verifyRetellSignature(
      body: string,
      signature: string | null,
      apiKey: string
    ): boolean {
      if (!signature) return false
      try {
        const expected = crypto.createHmac('sha256', apiKey).update(body).digest('hex')
        const sigBuffer = Buffer.from(signature, 'hex')
        const expectedBuffer = Buffer.from(expected, 'hex')
        if (sigBuffer.length !== expectedBuffer.length) return false
        return crypto.timingSafeEqual(sigBuffer, expectedBuffer)
      } catch {
        return false
      }
    }
    ```

    ```typescript
    function isAfterHours(
      businessHours: BusinessHours | null,
      timestampMs: number,
      timezone: string
    ): boolean {
      if (!businessHours) return false

      const dayNames = ['sun', 'mon', 'tue', 'wed', 'thu', 'fri', 'sat'] as const
      const date = new Date(timestampMs)
      const formatter = new Intl.DateTimeFormat('en-US', {
        timeZone: timezone,
        weekday: 'short',
        hour: '2-digit',
        minute: '2-digit',
        hour12: false,
      })

      const parts = formatter.formatToParts(date)
      const dayPart = parts.find(p => p.type === 'weekday')?.value?.toLowerCase()
      const hourStr = parts.find(p => p.type === 'hour')?.value ?? '00'
      const minStr = parts.find(p => p.type === 'minute')?.value ?? '00'

      const dayKey = dayNames.find(d => d === dayPart?.substring(0, 3))
      if (!dayKey) return false

      const hours = businessHours[dayKey]
      if (!hours || hours.closed) return true

      const current = parseInt(hourStr) * 60 + parseInt(minStr)
      const open = parseInt(hours.open.split(':')[0]) * 60 + parseInt(hours.open.split(':')[1])
      const close = parseInt(hours.close.split(':')[0]) * 60 + parseInt(hours.close.split(':')[1])

      return current < open || current >= close
    }
    ```
