    ```typescript
    import Anthropic from '@anthropic-ai/sdk'
    import { env } from '@/lib/utils/env'
    import type { LeadAnalysis } from '@/types/api'

    const anthropic = new Anthropic({ apiKey: env.anthropicApiKey() })

    const ANALYSIS_PROMPT = `Analyze this phone call transcript and extract lead information.

Return a JSON object with these exact fields:
{
  "is_lead": boolean,         // true if caller has a service need or buying intent
  "caller_name": string|null, // caller's name if mentioned
  "service_interested": string|null, // what service they called about
  "notes": string|null,       // important details about the caller's situation
  "urgency": "low"|"medium"|"high"|"urgent",
  "lead_score": number,       // integer 1-10: 9-10=urgent buyer, 7-8=qualified, 5-6=interested, 3-4=browsing, 1-2=not a lead
  "summary": string,          // 1-2 sentence plain summary of the call
  "sentiment": "positive"|"neutral"|"negative",
  "requires_callback": boolean // caller asked to be called back

Scoring guide:
- 9-10: Urgent need + contact info provided + ready to book
- 7-8: Clear service need + engaged conversation
- 5-6: Interest shown but no urgency or incomplete info
- 3-4: General inquiry, price checking, no commitment
- 1-2: Wrong number, hangup, or no service need

Transcript:
{{TRANSCRIPT}}`

    export async function analyzeCallTranscript(transcript: string): Promise<LeadAnalysis> {
      const message = await anthropic.messages.create({
        model: 'claude-3-5-haiku-20241022',  // Haiku — fast + cheap for extraction
        max_tokens: 400,
        messages: [{
          role: 'user',
          content: ANALYSIS_PROMPT.replace('{{TRANSCRIPT}}', transcript)
        }]
      })

      const text = message.content[0].type === 'text' ? message.content[0].text : ''
      const parsed = JSON.parse(text) as LeadAnalysis

      // Clamp and validate lead_score
      parsed.lead_score = Math.max(1, Math.min(10, Math.round(parsed.lead_score)))

      return parsed
    }
    ```

    ```typescript
    import Anthropic from '@anthropic-ai/sdk'
    import { env } from '@/lib/utils/env'

    const anthropic = new Anthropic({ apiKey: env.anthropicApiKey() })

    export async function scoreCallQuality(
      transcript: string,
      durationMs: number
    ): Promise<number> {
      // Short calls are low quality by definition
      if (durationMs < 10000) return 2  // < 10 seconds

      const message = await anthropic.messages.create({
        model: 'claude-3-5-haiku-20241022',
        max_tokens: 50,
        messages: [{
          role: 'user',
          content: `Rate the quality of this AI receptionist call on a scale of 1-10.
10 = AI successfully helped caller, captured info, professional
1 = AI confused, repeated itself, failed to help

Return ONLY a single integer, nothing else.

Transcript:
${transcript}`
        }]
      })

      const text = message.content[0].type === 'text' ? message.content[0].text.trim() : '5'
      const score = parseInt(text, 10)
      return isNaN(score) ? 5 : Math.max(1, Math.min(10, score))
    }
    ```
