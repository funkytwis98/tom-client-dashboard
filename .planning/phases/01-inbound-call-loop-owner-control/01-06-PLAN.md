    ```typescript
    import { createServiceClient } from '@/lib/supabase/service'
    import { retellClient } from './client'
    import type { Client, AgentConfig, KnowledgeEntry } from '@/types/domain'

    // Category display order in prompt
    const CATEGORY_ORDER = ['services', 'pricing', 'hours', 'policies', 'faq', 'location', 'team', 'promotions', 'competitors']

    export async function buildAgentPrompt(clientId: string): Promise<string> {
      const supabase = createServiceClient()

      const [{ data: client }, { data: knowledge }] = await Promise.all([
        supabase.from('clients').select('*, agent_config(*)').eq('id', clientId).single(),
        supabase.from('knowledge_base').select('*').eq('client_id', clientId).eq('is_active', true).order('priority', { ascending: false })
      ])

      if (!client) throw new Error(`Client ${clientId} not found`)

      const config: AgentConfig = client.agent_config?.[0] ?? {
        agent_name: 'receptionist',
        greeting: `Thanks for calling ${client.name}, how can I help you?`,
        personality: 'friendly, professional, and helpful',
        sales_style: 'consultative — ask good questions, understand the need, suggest solutions',
        escalation_rules: null,
        voicemail_message: `You've reached ${client.name}. We're currently closed. Please leave your name and phone number and we'll call you back shortly.`,
        voice_id: null,
        language: 'en-US',
        custom_instructions: null,
      }

      // Group knowledge by category
      const grouped: Record<string, KnowledgeEntry[]> = {}
      for (const entry of (knowledge ?? [])) {
        if (!grouped[entry.category]) grouped[entry.category] = []
        grouped[entry.category].push(entry)
      }

      // Build prompt sections
      const knowledgeSections = CATEGORY_ORDER
        .filter(cat => grouped[cat]?.length > 0)
        .map(cat => {
          const entries = grouped[cat]
          return `## ${cat.charAt(0).toUpperCase() + cat.slice(1)}\n${entries.map(e => `### ${e.title}\n${e.content}`).join('\n\n')}`
        })
        .join('\n\n')

      const hoursSection = client.business_hours
        ? formatBusinessHours(client.business_hours)
        : ''

      const escalation = config.escalation_rules ??
        "If caller is frustrated, angry, or has a complex complaint, say 'Let me have someone from our team call you right back.'"

      return `You are ${config.agent_name}, the AI receptionist for ${client.name}.

## Personality
${config.personality ?? 'Friendly, professional, and helpful'}

## Greeting
Always start the call with: "${config.greeting}"

${knowledgeSections}

${hoursSection}

## Your Sales Approach
${config.sales_style ?? 'Ask good questions. Understand what the caller needs. Suggest solutions.'}

## Escalation Rules
${escalation}

## Core Rules
- Always be helpful, warm, and professional
- If you don't know something, say "Let me have ${client.owner_name ?? 'someone from our team'} get back to you on that"
- Always try to get the caller's name and phone number
- Never make up information — only use what's in your knowledge above
- If the business is closed and caller asks for an appointment, offer to take their info and have someone call them back

${config.custom_instructions ? `## Additional Instructions\n${config.custom_instructions}` : ''}

## Information to Capture on Every Call
- Caller's name
- Best callback number
- What they need (service, question, complaint)
- Urgency — when do they need this done?`.trim()
    }

    function formatBusinessHours(hours: Record<string, { open: string; close: string; closed?: boolean }>): string {
      const days = { mon: 'Monday', tue: 'Tuesday', wed: 'Wednesday', thu: 'Thursday', fri: 'Friday', sat: 'Saturday', sun: 'Sunday' }
      const lines = Object.entries(days)
        .map(([key, label]) => {
          const h = hours[key]
          if (!h || h.closed) return `${label}: Closed`
          return `${label}: ${h.open} - ${h.close}`
        })
        .join('\n')
      return `## Business Hours\n${lines}`
    }

    export async function updateRetellAgent(clientId: string): Promise<void> {
      const supabase = createServiceClient()
      const { data: client } = await supabase
        .from('clients')
        .select('retell_agent_id, agent_config(*)')
        .eq('id', clientId)
        .single()

      if (!client?.retell_agent_id) {
        console.warn(`Client ${clientId} has no retell_agent_id — skipping agent sync`)
        return
      }

      const [systemPrompt, voiceId] = await Promise.all([
        buildAgentPrompt(clientId),
        Promise.resolve(client.agent_config?.[0]?.voice_id ?? null)
      ])

      await retellClient.agent.update(client.retell_agent_id, {
        agent_name: client.agent_config?.[0]?.agent_name ?? 'receptionist',
        response_engine: { type: 'retell-llm', llm_id: '' },  // preserved from existing
        ...(voiceId ? { voice_id: voiceId } : {}),
        llm_websocket_url: undefined,  // not applicable for retell-llm
      })

      // Note: system prompt is set via retellClient.llm.update — find the llm_id from the agent
      // For MVP: fetch agent details then update the associated LLM
      const agent = await retellClient.agent.retrieve(client.retell_agent_id)
      if (agent.response_engine.type === 'retell-llm') {
        const llmId = (agent.response_engine as { llm_id: string }).llm_id
        await retellClient.llm.update(llmId, { general_prompt: systemPrompt })
      }
    }
    ```

    ```typescript
    'use server'

    import { revalidatePath } from 'next/cache'
    import { createClient } from '@/lib/supabase/server'
    import { updateRetellAgent } from '@/lib/retell/agent-builder'
    import type { AgentConfig } from '@/types/domain'
    import { z } from 'zod'

    const AgentConfigSchema = z.object({
      client_id: z.string().uuid(),
      agent_name: z.string().min(1).max(50),
      greeting: z.string().max(500).nullable(),
      personality: z.string().max(1000).nullable(),
      sales_style: z.string().max(1000).nullable(),
      escalation_rules: z.string().max(1000).nullable(),
      voicemail_message: z.string().max(500).nullable(),
      voice_id: z.string().nullable(),
      language: z.string().default('en-US'),
      custom_instructions: z.string().max(2000).nullable(),
    })

    export async function updateAgentConfig(formData: z.infer<typeof AgentConfigSchema>) {
      const parsed = AgentConfigSchema.parse(formData)
      const supabase = await createClient()

      const { error } = await supabase
        .from('agent_config')
        .upsert({ ...parsed, updated_at: new Date().toISOString() }, { onConflict: 'client_id' })

      if (error) throw new Error(error.message)

      revalidatePath(`/clients/${parsed.client_id}/agent`)
      return { success: true }
    }

    export async function syncRetellAgent(clientId: string) {
      await updateRetellAgent(clientId)
      return { success: true }
    }
    ```

    Create `src/app/actions/knowledge.ts` — Server Actions for knowledge base CRUD:
    ```typescript
    'use server'

    import { revalidatePath } from 'next/cache'
    import { createClient } from '@/lib/supabase/server'
    import { z } from 'zod'

    const KnowledgeEntrySchema = z.object({
      id: z.string().uuid().optional(),
      client_id: z.string().uuid(),
      category: z.enum(['services', 'pricing', 'faq', 'hours', 'policies', 'location', 'team', 'promotions', 'competitors']),
      title: z.string().min(1).max(200),
      content: z.string().min(1).max(10000),
      priority: z.number().int().min(0).max(100).default(0),
      is_active: z.boolean().default(true),
    })

    export async function saveKnowledgeEntry(formData: z.infer<typeof KnowledgeEntrySchema>) {
      const parsed = KnowledgeEntrySchema.parse(formData)
      const supabase = await createClient()

      const { error, data } = parsed.id
        ? await supabase.from('knowledge_base').update({ ...parsed, updated_at: new Date().toISOString() }).eq('id', parsed.id).select().single()
        : await supabase.from('knowledge_base').insert(parsed).select().single()

      if (error) throw new Error(error.message)

      revalidatePath(`/clients/${parsed.client_id}/knowledge`)
      return { success: true, entry: data }
    }

    export async function deleteKnowledgeEntry(entryId: string, clientId: string) {
      const supabase = await createClient()
      // Soft delete — keeps history, doesn't break existing agent prompts mid-call
      const { error } = await supabase
        .from('knowledge_base')
        .update({ is_active: false, updated_at: new Date().toISOString() })
        .eq('id', entryId)

      if (error) throw new Error(error.message)

      revalidatePath(`/clients/${clientId}/knowledge`)
      return { success: true }
    }

    export async function reorderKnowledgeEntries(
      entries: Array<{ id: string; priority: number }>,
      clientId: string
    ) {
      const supabase = await createClient()

      // Bulk update priorities
      const updates = entries.map(({ id, priority }) =>
        supabase.from('knowledge_base').update({ priority }).eq('id', id)
      )

      await Promise.all(updates)

      revalidatePath(`/clients/${clientId}/knowledge`)
      return { success: true }
    }
    ```
